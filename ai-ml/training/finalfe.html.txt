        // Theme Toggle
        function toggleTheme() {
            currentTheme = currentTheme === 'light' ? 'dark' : 'light';
            document.documentElement.setAttribute('data-theme', currentTheme);
            document.getElementById('themeIcon').textContent = currentTheme === 'light' ? '🌙' : '☀️';
            localStorage.setItem('theme', currentTheme);
            
            // Update charts
            if (window.dashboardChart) {
                updateChartTheme(window.dashboardChart);
            }
            if (window.analyticsChart) {
                updateChartTheme(window.analyticsChart);
            }
            if (window.comparisonChart) {
                updateChartTheme(window.comparisonChart);
            }
        }

        function updateChartTheme(chart) {
            const textColor = getComputedStyle(document.documentElement).getPropertyValue('--text-primary').trim();
            const gridColor = getComputedStyle(document.documentElement).getPropertyValue('--border-color').trim();
            
            if (chart.options.scales) {
                if (chart.options.scales.x) {
                    chart.options.scales.x.ticks.color = textColor;
                    chart.options.scales.x.grid.color = gridColor;
                }
                if (chart.options.scales.y) {
                    chart.options.scales.y.ticks.color = textColor;
                    chart.options.scales.y.grid.color = gridColor;
                }
                if (chart.options.scales.r) {
                    chart.options.scales.r.ticks.color = textColor;
                    chart.options.scales.r.grid.color = gridColor;
                    chart.options.scales.r.pointLabels.color = textColor;
                }
            }
            
            if (chart.options.plugins && chart.options.plugins.legend) {
                chart.options.plugins.legend.labels.color = textColor;
            }
            
            chart.update();
        }

        function getMyLocation() {
            if (!map) {
                alert('Map is not initialized. Please wait or refresh the page.');
                return;
            }

            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(position => {
                    const lat = position.coords.latitude;
                    const lng = position.coords.longitude;
                    
                    if (userMarker) {
                        map.removeLayer(userMarker);
                    }
                    
                    userMarker = L.marker([lat, lng]).addTo(map);
                    userMarker.bindPopup('<b>Your Location</b>').openPopup();
                    map.setView([lat, lng], 13);
                }, error => {
                    alert('Unable to get your location. Please enable location services.');
                });
            } else {
                alert('Geolocation is not supported by your browser.');
            }
        }

            if (!map || !userMarker) {

            // Clear existing water body markers
            clearWaterBodies();

            // Simulate API call with timeout
            setTimeout(() => {
                // Generate mock water bodies near user location
                const waterBodies = generateMockWaterBodies(userLatLng.lat, userLatLng.lng);

        function generateMockWaterBodies(userLat, userLng) {
            const waterBodyTypes = ['river', 'lake', 'pond', 'well', 'reservoir'];
            const waterBodies = [];
            
            // Generate 3-6 random water bodies near user location
            const count = Math.floor(Math.random() * 4) + 3;
            
            for (let i = 0; i < count; i++) {
                // Generate random coordinates within 5km of user
                const lat = userLat + (Math.random() - 0.5) * 0.05;
                const lng = userLng + (Math.random() - 0.5) * 0.05;
                
                const type = waterBodyTypes[Math.floor(Math.random() * waterBodyTypes.length)];
                const ph = (Math.random() * 2 + 6.5).toFixed(1); // pH between 6.5-8.5
                const turbidity = (Math.random() * 10).toFixed(1); // Turbidity between 0-10 NTU
                
                // Determine quality based on pH and turbidity
                let quality = 'Good';
                if (ph < 6.5 || ph > 8.5 || turbidity > 5) {
                    quality = 'Moderate';
                }
                if (ph < 6.0 || ph > 9.0 || turbidity > 8) {
                    quality = 'Poor';
                }
                
                waterBodies.push({
                    id: i,
                    name: `${translations[currentLanguage][type] || type} ${i+1}`,
                    type: type,
                    lat: lat,
                    lng: lng,
                    ph: ph,
                    turbidity: turbidity,
                    quality: quality
                });
            }
            
            return waterBodies;
        }

        function displayWaterBodies(waterBodies, waterBodyGrid) {
            waterBodyGrid.innerHTML = '';
            
            if (waterBodies.length === 0) {
                waterBodyGrid.innerHTML = `<div class="loading">${translations[currentLanguage].noWaterBodies || 'No water bodies found in your area'}</div>`;
                return;
            }
            
            waterBodies.forEach(waterBody => {
                // Add marker to map
                const color = waterBody.quality === 'Good' ? 'green' : waterBody.quality === 'Moderate' ? 'orange' : 'red';
                const marker = L.circleMarker([waterBody.lat, waterBody.lng], {
                    radius: 10,
                    fillColor: color,
                    color: '#fff',
                    weight: 2,
                    opacity: 1,
                    fillOpacity: 0.8
                }).addTo(map);
                
                marker.bindPopup(`
                    <b>${waterBody.name}</b><br>
                    ${translations[currentLanguage].ph || 'pH'}: ${waterBody.ph}<br>
                    ${translations[currentLanguage].turbidity || 'Turbidity'}: ${waterBody.turbidity} NTU<br>
                    ${translations[currentLanguage].quality || 'Quality'}: ${waterBody.quality}
                `);
                
                waterBodyMarkers.push(marker);

                    <h4>${waterBody.name}</h4>

            });
        }

        function loadQuestion() {
            if (currentQuestion >= quizData.length) {
                showResults();
                return;

            const question = quizData[currentQuestion];
            document.getElementById('quizQuestion').textContent = `Question ${currentQuestion + 1}: ${question.question}`;
            document.getElementById('quizScore').textContent = `Score: ${score} / ${answeredQuestions}`;
            document.getElementById('progressFill').style.width = `${(currentQuestion / quizData.length) * 100}%`;
            
            const optionsContainer = document.getElementById('quizOptions');
            optionsContainer.innerHTML = '';
            
            question.options.forEach((option, index) => {
                const optionDiv = document.createElement('div');
                optionDiv.className = 'quiz-option';
                optionDiv.textContent = option;
                optionDiv.onclick = () => selectAnswer(index);
                optionsContainer.appendChild(optionDiv);
            });

            document.getElementById('quizExplanation').classList.remove('show');
            document.getElementById('nextBtn').style.display = 'none';
        }

        function selectAnswer(selectedIndex) {

            options.forEach((option, index) => {
                option.onclick = null;
                if (index === question.correct) {
                    option.classList.add('correct');
                }
                if (index === selectedIndex && index !== question.correct) {
                    option.classList.add('incorrect');
                }
            });

            answeredQuestions++;
            if (selectedIndex === question.correct) {
                score++;
            } else {
                // Show explanation for wrong answer
                const explanationDiv = document.getElementById('quizExplanation');
                explanationDiv.innerHTML = `<strong>Explanation:</strong> ${question.explanation}`;
                explanationDiv.classList.add('show');
            }

            document.getElementById('quizScore').textContent = `Score: ${score} / ${answeredQuestions}`;
            document.getElementById('nextBtn').style.display = 'block';
        }

        function nextQuestion() {
            currentQuestion++;
            loadQuestion();
        }

        function showResults() {
            const percentage = (score / quizData.length) * 100;
            let message = '';
            
            if (percentage >= 80) {
                message = 'Excellent! You have great knowledge about water quality!';
            } else if (percentage >= 60) {
                message = 'Good job! Keep learning about water quality.';
            } else {
                message = 'Keep practicing! Review the materials to improve your knowledge.';
            }

            document.getElementById('quizQuestion').textContent = `Quiz Complete!`;
            document.getElementById('quizOptions').innerHTML = `
                <div style="text-align: center; padding: 40px;">
                    <h2>Your Score: ${score} / ${quizData.length}</h2>
                    <h3>${percentage.toFixed(1)}%</h3>
                    <p style="margin-top: 20px; font-size: 18px;">${message}</p>
                </div>
            `;
            document.getElementById('nextBtn').style.display = 'none';
            document.getElementById('restartBtn').style.display = 'block';
            document.getElementById('progressFill').style.width = '100%';
        }

        function restartQuiz() {
            currentQuestion = 0;
            score = 0;
            answeredQuestions = 0;
            document.getElementById('restartBtn').style.display = 'none';
            loadQuestion();
        }

        // Reports & Complaints Functions
        function initReports() {
            // Load complaints from localStorage
            const savedComplaints = localStorage.getItem('waterComplaints');
            if (savedComplaints) {
                complaints = JSON.parse(savedComplaints);
            }
            
            // Set up form submission
            document.getElementById('reportForm').addEventListener('submit', function(e) {
                e.preventDefault();
                submitReport();
            });
            
            // Initial load of complaints
            refreshComplaints();
        }

        function submitReport() {
            const issueType = document.getElementById('issueType').value;
            const location = document.getElementById('location').value;
            const description = document.getElementById('description').value;
            const priority = document.getElementById('priority').value;
            const contact = document.getElementById('contact').value;
            
            // Create new complaint
            const newComplaint = {
                id: Date.now(), // Use timestamp as unique ID
                type: issueType,
                location: location,
                description: description,
                priority: priority,
                status: "pending",
                date: new Date().toISOString().split('T')[0],
                contact: contact
            };
            
            // Add to complaints array
            complaints.unshift(newComplaint);
            
            // Save to localStorage
            localStorage.setItem('waterComplaints', JSON.stringify(complaints));
            
            // Show success message
            alert(translations[currentLanguage].reportSubmitted || 'Report submitted successfully!');
            
            // Reset form
            document.getElementById('reportForm').reset();
            
            // Refresh complaints list
            refreshComplaints();
        }

        function refreshComplaints() {
            const complaintsList = document.getElementById('complaintsList');
            
            if (complaints.length === 0) {
                complaintsList.innerHTML = `
                    <div class="empty-state">
                        <div class="empty-state-icon">📝</div>
                        <h3>${translations[currentLanguage].noComplaints || 'No complaints found'}</h3>
                        <p>Submit a report using the form on the left to get started.</p>
                    </div>
                `;
                return;
            }
            
            complaintsList.innerHTML = '';
            
            complaints.forEach(complaint => {
                const complaintCard = document.createElement('div');
                complaintCard.className = `complaint-card ${complaint.status === 'resolved' ? 'resolved' : ''} ${complaint.priority === 'critical' ? 'critical' : ''}`;
                
                // Get status text based on current language
                let statusText = '';
                if (complaint.status === 'pending') {
                    statusText = translations[currentLanguage].statusPending || 'Pending';
                } else if (complaint.status === 'in-progress') {
                    statusText = translations[currentLanguage].statusInProgress || 'In Progress';
                } else {
                    statusText = translations[currentLanguage].statusResolved || 'Resolved';
                }
                
                // Get type display text
                const typeText = getTypeDisplayText(complaint.type);
                
                complaintCard.innerHTML = `
                    <div class="complaint-header">
                        <div>
                            <div class="complaint-title">${typeText} - ${complaint.location}</div>
                            <div class="complaint-meta">
                                <span>${complaint.date}</span>
                                <span>Priority: ${complaint.priority}</span>
                            </div>
                        </div>
                        <span class="complaint-status status-${complaint.status}">${statusText}</span>
                    </div>
                    <div class="complaint-description">${complaint.description}</div>
                    ${complaint.contact ? `<div class="complaint-meta">Contact: ${complaint.contact}</div>` : ''}
                    <div class="complaint-actions">
                        ${complaint.status !== 'resolved' ? 
                            `<button class="action-btn resolve-btn" onclick="resolveComplaint(${complaint.id})">
                                ${translations[currentLanguage].markResolved || 'Mark Resolved'}
                            </button>` : ''
                        }
                        <button class="action-btn delete-btn" onclick="deleteComplaint(${complaint.id})">
                            ${translations[currentLanguage].deleteReport || 'Delete'}
                        </button>
                    </div>
                `;
                
                complaintsList.appendChild(complaintCard);
            });
        }

        function getTypeDisplayText(type) {
            const typeMap = {
                'water_quality': 'Water Quality Issue',
                'contamination': 'Water Contamination',
                'infrastructure': 'Infrastructure Problem',
                'supply': 'Water Supply Issue',
                'other': 'Other Issue'
            };
            return typeMap[type] || 'Unknown Issue';
        }

        function resolveComplaint(id) {
            const complaintIndex = complaints.findIndex(c => c.id === id);
            if (complaintIndex !== -1) {
                complaints[complaintIndex].status = 'resolved';
                localStorage.setItem('waterComplaints', JSON.stringify(complaints));
                refreshComplaints();
            }
        }

        function deleteComplaint(id) {
            if (confirm('Are you sure you want to delete this report?')) {
                complaints = complaints.filter(c => c.id !== id);
                localStorage.setItem('waterComplaints', JSON.stringify(complaints));
                refreshComplaints();
            }
        }
